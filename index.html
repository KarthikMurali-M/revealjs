Code: EBITDA and Margin Calculation

# Example Python snippet to compute EBITDA margin
revenue = 1240_000_000  # in USD
operating_income = 315_000_000
depreciation = 25_000_000
amortization = 10_000_000

ebitda = operating_income + depreciation + amortization
ebitda_margin = ebitda / revenue

print(f"EBITDA: ${ebitda:,}")
print(f"EBITDA Margin: {ebitda_margin:.2%}")
